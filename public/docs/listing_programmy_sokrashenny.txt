    35  def hash_password(password: str, salt: str = None) -> str:
    36      if salt is None:
    37          salt = base64.urlsafe_b64encode(os.urandom(16)).decode()
    38      digest = hashlib.pbkdf2_hmac('sha256', password.encode(), salt.encode(), 100000)
    39      return f"{salt}${base64.urlsafe_b64encode(digest).decode()}"

    42  def verify_password(password: str, stored: str) -> bool:
    43      try:
    44          salt, _ = stored.split('$', 1)
    45          candidate = hash_password(password, salt)
    46          return hmac.compare_digest(candidate, stored)
    47      except Exception:
    48          return False

    51  def make_token(payload: dict, secret: str, ttl_seconds: int = 60 * 60 * 24 * 30) -> str:
    52      header = {'alg': 'HS256', 'typ': 'JWT'}
    53      payload = dict(payload)
    54      payload['exp'] = int(time.time()) + ttl_seconds
    55      segments = [
    56          _b64url_encode(json.dumps(header).encode()),
    57          _b64url_encode(json.dumps(payload).encode()),
    58      ]
    59      signing_input = '.'.join(segments).encode()
    60      signature = hmac.new(secret.encode(), signing_input, hashlib.sha256).digest()
    61      segments.append(_b64url_encode(signature))
    62      return '.'.join(segments)

    65  def verify_token(token: str, secret: str):
    66      try:
    67          header_b64, payload_b64, sig_b64 = token.split('.')
    68          signing_input = f"{header_b64}.{payload_b64}".encode()
    69          expected_sig = hmac.new(secret.encode(), signing_input, hashlib.sha256).digest()
    70          actual_sig = _b64url_decode(sig_b64)
    71          if not hmac.compare_digest(expected_sig, actual_sig):
    72              return None
    73          payload = json.loads(_b64url_decode(payload_b64))
    74          if payload.get('exp', 0) < time.time():
    75              return None
    76          return payload
    77      except Exception:
    78          return None
